function showScreen(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  const target = document.getElementById('screen-' + id);
  if (target) {
    target.classList.add('active');
    target.scrollTop = 0;
  }
  document.querySelectorAll('.screen-nav-btn').forEach(b => b.classList.remove('active'));
  document.querySelectorAll('.screen-nav-btn').forEach(b => {
    if (b.getAttribute('onclick') === `showScreen('${id}')`) b.classList.add('active');
  });
  const counter = document.getElementById('screenCounter');
  if (counter) counter.textContent = screenNames[id] || id;
}
